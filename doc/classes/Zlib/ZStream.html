avail_out=(size)

Allocates size bytes in output buffer. If size < avail_out it truncates the buffer.

[ show source ]

# File lib/zliby.rb, line 40
        def avail_out= size
                size.times do 
                        if size > avail_out
                                @output_buffer.push nil
                        else
                                @output_buffer.pop
                        end
                end
        end
